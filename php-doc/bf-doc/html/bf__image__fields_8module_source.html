176 function bf_image_fields_field_attach_update($entity_type, $entity) {
177  list($id, $vid, $bundle) = entity_extract_ids($entity_type, $entity);
178 
179  if (!isset($vid)) {
180  $vid = $id;
181  }
182 
183  $image_fields = bf_image_fields_get_image_fields($entity_type, $bundle);
184 
185  foreach ($image_fields as $field_name => $image_field) {
186  if (!$image_field['settings']['image_field_caption']) {
187  continue;
188  }
189  $field = field_info_field($field_name);
190  $table_name = 'field_image_field_caption';
191  $revision_name = 'field_image_field_caption_revision';
192 
193  $all_languages = field_available_languages($entity_type, $field);
194  $field_languages = array_intersect($all_languages, array_keys((array) $entity->$field_name));
195 
196  // Delete and insert, rather than update, in case a value was added.
197  // Delete languages present in the incoming $entity->$field_name.
198  // Delete all languages if $entity->$field_name is empty.
199  $languages = !empty($entity->$field_name) ? $field_languages : $all_languages;
200  if ($languages) {
201  db_delete($table_name)
202  ->condition('field_name', $field_name)
203  ->condition('entity_type', $entity_type)
204  ->condition('entity_id', $id)
205  ->condition('language', $languages, 'IN')
206  ->execute();
207  db_delete($revision_name)
208  ->condition('field_name', $field_name)
209  ->condition('entity_type', $entity_type)
210  ->condition('entity_id', $id)
211  ->condition('revision_id', $vid)
212  ->condition('language', $languages, 'IN')
213  ->execute();
214  }
215 
216  // Prepare the multi-insert query.
217  $do_insert = FALSE;
218  $columns = array('field_name', 'entity_type', 'entity_id', 'revision_id', 'bundle', 'delta', 'language', 'caption', 'caption_url');
219  $query = db_insert($table_name)->fields($columns);
220  $revision_query = db_insert($revision_name)->fields($columns);
221 
222  foreach ($field_languages as $langcode) {
223 
224  $items = (array) $entity->{$field_name}[$langcode];
225  $delta_count = 0;
226  foreach ($items as $delta => $item) {
227  // We now know we have something to insert.
228  $do_insert = TRUE;
229  if (!isset($item['image_field_caption'])) {
230  $caption = '';
231  } else {
232  $caption = $item['image_field_caption'];
233  }
234  if (!isset($item['image_field_caption_url'])) {
235  $caption_url = '';
236  } else {
237  $caption_url = $item['image_field_caption_url'];
238  }
239 
240  $record = array(
241  'field_name' => $field_name,
242  'entity_type' => $entity_type,
243  'entity_id' => $id,
244  'revision_id' => $vid,
245  'bundle' => $bundle,
246  'delta' => $delta,
247  'language' => $langcode,
248  'caption' => $caption,
249  'caption_url' => $caption_url,
250  );
251  $query->values($record);
252  if (isset($vid)) {
253  $revision_query->values($record);
254  }
255 
256  if ($field['cardinality'] != FIELD_CARDINALITY_UNLIMITED && ++$delta_count == $field['cardinality']) {
257  break;
258  }
259  }
260  }
261 
262  // Execute the query if we have values to insert.
263  if ($do_insert) {
264  $query->execute();
265  $revision_query->execute();
266  }
267  }
