189  //checking for removed users
190  $user = user_load($editor['target_id']);
191  $original_editors[$user->uid] = $user;
192  if(empty($editors[$user->uid])){
193  $removed_editors[] = empty($user->field_displayed_name[LANGUAGE_NONE][0]['value'])?$user->name:$user->field_displayed_name[LANGUAGE_NONE][0]['value'];
194  }

196  }
197  if(!empty($field)){
198  foreach ($field as $editor){
199  //checking for added users

201  $added_editors[] = empty($editors[$editor['target_id']]->field_displayed_name[LANGUAGE_NONE][0]['value'])?$editors[$editor['target_id']]->name:$editors[$editor['target_id']]->field_displayed_name[LANGUAGE_NONE][0]['value'];
202  }

204  }
205  }
206  foreach ($tokens as $name => $original) {

210  break;
211  case 'editors-added':
