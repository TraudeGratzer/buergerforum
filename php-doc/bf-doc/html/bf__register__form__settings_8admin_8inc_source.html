205 function bf_register_form_settings_validate(&$form, &$form_state) {
206  $values = array();
207  $textValues = '';
208  $textName = '';
209  $error = FALSE;
210  foreach ($form_state['input'] as $keyparam => $param) {
211  if (preg_match('/allowed_values_/', $keyparam)) {
212  if ($param) {
213  $list = explode("\n", $param);
214  $list = array_map('trim', $list);
215  $list = array_filter($list, 'strlen');
216  foreach ($list as $keylist => $value) {
217  // Check for a manually specified key.
218  if (strpos($value, '|') !== FALSE) {
219  list($keylist, $value) = explode('|', $value);
220  }
221  // Otherwise see if we need to use the value as the key. The "list" type
222  // will automatically convert non-keyed lines to integers.
223  else {
224  $keylist = $value;
225  }
226  $values[$keylist] = (isset($value) && $value !== '') ? $value : $keylist;
227  }
228  }
229  else {
230  $error = TRUE;
231  $textValues = $param;
232  }
233  }
234  if (preg_match('/name_/', $keyparam)) {
235 
236  if (!$param) {
237  $error = TRUE;
238  }
239  else {
240  $textName = $param;
241  }
242  }
243  }
244  if ($error) {
245  bf_register_form_create_fields($form, $form_state['count_fields'], $textName, $textValues);
246  if (!$textName && !$textValues) {
247  form_set_error('manager_settings', t('Empty field name and empty field values.'));
248  }
249  elseif (!$textName) {
250  form_set_error('manager_settings', t('Empty field name.'));
251  }
252  else {
253  form_set_error('manager_settings', t('Empty field values.'));
254  }
255  }
256  else {
257  $form_state['values']['finalValues'] = $values;
258  $form_state['values']['finalName'] = $textName;
259  }
