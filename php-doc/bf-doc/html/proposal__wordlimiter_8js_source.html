228  ml.truncate_html = function (text, limit, end) {
229  // The html result after cut.
230  var result_html = '';
231  // The text result, that will actually used when counting characters.
232  var result_text = '';
233  // A stack that will keep the tags that are open at a given time.
234  var tags_open = new Array();
235  // making the lineendings with two chars
236  // alert(end);
237  /* if (!end) {
238  text = ml.twochar_lineending(text);
239  }*/
240 
241  // We do want that the space characters to count as 1, not 6...
242  text = text.replace('&nbsp;', ' ');
243  while (result_text.length < limit && text.length > 0) {
244  switch (text.charAt(0)) {
245  case '<':
246  {
247  if (text.charAt(1) != '/') {
248  var tag_name = '';
249  var tag_name_completed = false;
250  while (text.charAt(0) != '>' && text.length > 0) {
251  var first_char = text.charAt(0).toString();
252  // Until the tag is closed, we do not append anything
253  // to the visible text, only to the html.
254  result_html += first_char;
255  // Also, check if we have a valid tag name.
256  if (!tag_name_completed && first_char == ' ') {
257  // We have the tag name, so push it into the stack.
258  tag_name_completed = true;
259  tags_open.push(tag_name);
260  }
261  // Check if we are still in the tag name.
262  if (!tag_name_completed && first_char != '<') {
263  tag_name += first_char;
264  }
265  //If we have the combination "/>" it means that the tag
266  //is closed, so remove it from the open tags stack.
267  if (first_char == '/' && text.length > 1 && text.charAt(1) == '>') {
268  tags_open.pop();
269  }
270  //Done with this char, remove it from the original text.
271  text = text.substring(1);
272  }
273  if (!tag_name_completed) {
274  // In this case we have a tag like "<strong>some text</strong> so
275  // we did not have any attributes in the tag, but still, the tag
276  // has to be marked as open.
277  tags_open.push(tag_name);
278  }
279  //We are here, then the tag is closed, so just remove the
280  //remaining ">" character.
281  if (text.length > 0) {
282  result_html += text.charAt(0).toString();
283  }
284  } else {
285  // In this case, we have an ending tag.
286  // The name of the ending tag should match the last open tag,
287  // otherwise, something is wrong with th html text.
288  var tag_name = '';
289  while (text.charAt(0) != '>' && text.length > 0) {
290  var first_char = text.charAt(0).toString();
291  if (first_char != '<' && first_char != '/') {
292  tag_name += first_char;
293  }
294  result_html += first_char;
295  text = text.substring(1);
296  }
297  if (text.length > 0) {
298  result_html = result_html + text.charAt(0).toString();
299  }
300  // Pop the last element from the tags stack and compare it with
301  // the tag name.
302  var expected_tag_name = tags_open.pop();
303  if (expected_tag_name != tag_name) {
304  //Should throw an exception, but for the moment just alert.
305  alert('Expected end tag: ' + expected_tag_name + '; Found end tag: ' + tag_name);
306  }
307  }
308  break;
309  }
310  default:
311  {
312  // In this case, we have a character that should also count for the
313  // limit, so append it to both, the html and text result.
314  var first_char = text.charAt(0).toString();
315  result_html += first_char;
316  result_text += first_char;
317  break;
318  }
319  }
320  // Remove the first character, it did its job.
321  text = text.substring(1);
322  }
323  // Restore the open tags that were not closed. This happens when the text
324  // got truncated in the middle of one or more html tags.
325  var tag = '';
326  while (tag = tags_open.pop()) {
327  result_html += '</' + tag + ">";
328  }
329  return result_html;
330  }
